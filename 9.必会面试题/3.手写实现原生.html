# promise

class MyPromise {
    constructor(cbFn) {
        /* 有三种状态：PENDING、FULFILLED、REJECTED */
        this.status = 'PENDING';
        /* 传递到then的值 */
        this.value = undefined;
        /* 成功状态的回调函数集合 */
        this.fulfilledFns = [];
        /* 失败状态的回调函数集合 */
        this.rejectedFns = [];

        let resolveFn = result => {
            let timer = setTimeout(() => {
                clearTimeout(timer);
                if(this.status === 'PENDING') return;
                this.status = 'FULFILLED';
                this.value = result;
                this.fulfilledFns.forEach(fulfilledFn => fulfilledFn(this.value));
            });
        }
        let rejectFn = reason => {
            let timer = setTimeout(() => {
                clearTimeout(timer);
                if(this.status !== 'PENDING') return;
                this.status = 'REJECTED';
                this.value = reason;
                this.rejectedFns.forEach(rejectedFn => rejectedFn(this.value));
            })
        }
        try {
            cbFn(resolveFn, rejectFn);
        }
    }
    then(fulfilledCb, rejectedCb) {
        typeof fulfilledCb !== 'function' ? fulfilledCb = result => result : null;
        typeof rejectedCb !== 'function' ? rejectedCb => reason => {
            throw new Error(reason typeof Error ? reason.message : reason);
        } : null;
        return new MyPromise((resolve, reject) => {
            this.fulfilledFns.push(() => {
                try {
                    let x = fulfilledCb(this.value);
                    x instanceof MyPromise ? x.then(resolve, reject) : resolve(x);
                } catch(err) {
                    reject(err);
                }
            });
            this.rejectedFns.push(() => {
                try {
                    let x = rejectedCb(this.value);
                    x instanceof MyPromise ? x.then(resolve, reject) : reject(x);
                } catch(err) {
                    reject(err);
                }
            })
        })
    }
    catch(rejectedCb) {
        return this.then(null, rejectedCb);
    }
    static all(promiseFns = []) {
        return new Promise((resolve, reject) => {
            let index = 0, result = [], len = promiseFns.length;
            for(let i = 0; i < len; i++) {
                promiseFns[i].then(val => {
                    index++;
                    result[i] = val;

                    }
                }, reject);
            }
        })
    }
}
