# 2.节流和防抖

# 节流throttle

一定时间内执行一次

const throttle = (fn, interval) => {
    let flag = true;
    return function() {
        if(flag) {
            flag = false;
            setTimeout(() => {
                fn();
                flag = true;
            })
        }
    }
}

# 防抖debounce

在delay时间内没有再次点击，则执行；

const debounce = (fn, delay) => {
    let timer = null;
    return function() {
        if(timer) {
            clearTimeout(timer);
            timer = setTimeout(fn, delay);
        } else {
            timer = setTimeout(fn, delay);
        }
    }
}
