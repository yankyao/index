# 12.设计模式

设计模式在一定程度上都降低了JS对象之间耦合度

不同的业务场景适用不同的设计模式

TIP

前端面试主要考察：单例模式、观察者模式、发布订阅模式

# 单例模式

一个类仅有一个实例，并提供一个访问它的全局入口。vuex经常就是一个单例，一个全局的状态管理，只提供一个接口访问。

function Singleton(name) {
    this.name = name;
}
Singleton.prototype.getName = function() {
    console.log(this.name);
}
Singleton.getInstance = (function() {
    let instance = null;
    return function(name) {
        if(!instance) {
            instance = new Singleton(name);
        }
        return instance;
    }
})();
const a = Singleton.getInstance();
const b = Singleton.getInstance();
console.log(a===b);
/* es6 */
const Singleton = (function() {
    let instance = null;
    return class _Singleton {
        constructor(name) {
            this.name = name;
            if(!instance) {
                instance = this;
            }
            return instance; // return就不会继续执行创建实例的四步走
        }
    }
})();
const a = new Singleton();
const b = new Singleton();
console.log(a===b);

# 观察者模式

定义了对象间一种一对多的依赖关系，当目标对象Subject的状态发生改变时，所有依赖它的对象Observer都会得到通知。

// 观察者对象
class Subject{
    constructor() {
        this.observers = [];
    }
    add(observer) {
        this.observers.push(observer);
        return this;
    }
    remove(observer) {
        const idx = this.observers.findIndex(i => i === observer);
        idx > -1 && this.observers.splice(idx, 1);
        return this;
    }
    notify() {
        for(let observer of this.observers) {
            observer.update();
        }
    }
}
// 观察者
class Observer {
    constructor(name) {
        this.name = name;
    }
    update() {
        console.log(`${this.name}更新了`);
    }
}

# 发布订阅模式

属于观察者模式的一种

  • 优点： 时间上、对象之间的解耦。
  • 缺点：消耗一定的时间和内存，订阅的处理函数不一定会被执行，驻留内存有性能开销； 弱化了对象间的联系，复杂的情况下可能会导致程序难以跟踪维护和理解；
const eventEmitter = {
    cache: {},
    on(event, fn) {
        !this.cache[event] && this.cache[event] = [];
        this.cache[event].push(fn);
        return this;
    },
    once(event, fn) {
        const wrapFn = (..args) => {
            this.off(event, wrapFn);
            fn.apply(this, args)
        }
        this.on(event, wrapFn);
        return this;
    },
    off(event, fn) {
        const list = this.cache[event];
        if(!list) return false;
        if(!fn) {
            this.cache[event] = []
        } else {
            const idx = list.findIndex(i => i === fn);
            if(idx !== -1) {
                list.splice(idx, i);
            }
        }
        return this;
    },
    emit(event, ...args) {
        const event = args[0];
        const fns = this.cache[event];
        if(!fns || fns.length === 0) return;
        fns.forEach(fn => {
            fn.applu(this, args)
        })
        return this;
    },

}

# 代理模式

# 策略模式

其核心是将算法的实现和算法的使用分离开来；一般包含两个部分：

  • 策略类，封装具体的算法，并负责具体的执行过程；
  • 环境类，把请求委托给某一个策略(算法)；
/* 简单的 */
const bonusTypes = {
    s (salary) {
        return salary * 5
    },
    a (salary) {
        return salary * 3
    },
    b (salary) {
        return salary * 1
    },
}
const calcBouns = (level, salary) => {
    return bonusTypes[level](salary);
}
console.log(calcBouns('a', 30000)); // 90000
/* es6，可扩展性强的 */
class Level {
    constructor(base) {
        this.base = base || 0;
    } 
    calc(salary) {
        return this.base * salary;
    }
}
class Bonus {
    constructor() {
        this.salary = null;
        this.level = null;
    } 
    setSalary(salary) {
        this.salary = salary;
    }
    setLevel(level) {
        this.level = level;
    }
    getBonus() {
        return this.level.calc(this.salary);
    }
}
const bonus = new Bonus();
bonus.setSalary(30000);
bonus.setLevel(new Level(5));
bonus.getBonus();

# 命令模式

命令模式的本质是对命令进行封装，将发出命令的责任和执行命令的责任分割开;

class Command1{
    constructor(receiver) {
        this.receiver = receiver;
    }
    execte() {
        this.receiver.action1();
    }
}
class ActionA{
    action1() {
        console.log('执行一些A1操作');
    }
    action2() {
        console.log('执行一些A2操作');
    }
}
class ActionB{
    action1() {
        console.log('执行一些B1操作');
    }
    action2() {
        console.log('执行一些B2操作');
    }
}
class Invoker{
    constructor(actionA, actionB) {
        this.actionA = actionA;
        this.actionB = actionB;
    }
    executeA() {
        this.actionA.execte();
    }
    executeB() {
        this.actionB.execte();
    }
}
// 使用
const commandA1 = new Command1(new ActionA());
const commandB1 = new Command1(new ActionB());
const invoker = new Invoker(commandA1, commandB1);
invoker.executeA();
invoker.executeB();
