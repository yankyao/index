# 14.TS

TypeScript是JavaScript的超集！ 作为前端2020面一定要熟练掌握TS！ TypeScript在编译成JavaScript的过程中，多了静态类型检查。

# 数据类型

ts中有以下数据类型：布尔boolean、数字number、字符串string、元组tuple、枚举enum、任意any、null、undefined、void、never

// 字符串
const num:string = '1';
// object
const a:object = {};
// 数组
const arr1:number[]=[1,2];
const arr2:Array<number> = [3,4];
// 数组的一种，元组类型
const arr3:[string, number] = ['2', 1]
// 枚举
enum flag {
    success=1,
    error=-1
}
const f:flag=flag.success;
enum color {red, green=5, blue}
const red:color=color.red; // 0，不赋值默认是索引值；
const green:color=color.green; // 5
const blue:color=color.blue; // 6
// any
let b:any = 1;
b = 'string'; // 可以再被赋值其他类型数据
// null undefined 是never的子类型
// let num2:number;
// console.log(num2); // 编译报错
let num2:undefined;
console.log(num2); // 编译通过
let n:null=null;
// ｜ 进行类型校验，未赋值则为undefined
let num3:number | string ｜ undefined;
// void 一般用于定义方法时，方法没有返回值的情况；
function run():void {
    console.log(1);
}
// never：只能被never类型赋值，包括undefined、null;
let n2:null;
n2=null;

var n3:never;
n3=(() => {
    throw new Error('error');
})();

# 函数

定义返回值，参数类型校验，可选参数，剩余参数的写法；

// 定义返回值
function fun():string {
    return '';
}

const fun2 = function ():number {
    return 123;
}
// 定义参数类型
// ts中实参和形参必须一致，或者配置可选参数或者给参数设置默认值；
// 可选参数建议，写到形参的最后面；
function fun3(p1:string,  p3:number=10, p2?:number):string {
    return `${p1}----${p2}`;
}
fun3('1');
fun3('1', 3); // 编译的时候都不会报错
function fun4():void {} // 没有返回值
// 剩余参数
function fun5(a:number,b:number,c:number,d:number):number {
    return a+b+c+d;
}
// 等于，剩余参数
function fun6(...result:number[]):number {
    return result.reduce((res, i) => res + i);
}
// 函数的重载，js同名函数以最后一个为准，ts则支持重载
function fun7(a:string):string;
function fun7(b:string):string {
    return b
}

# 类和静态方法

es5：

function Person(name, age){
    this.name = name;
    this.age = age;
    /* 实例之间不共享 */
    this.run = function() {
        console.log(this.name + 'run');
    }
}
/* 静态属性 */
Person.name = 'Person';
/* 静态方法 */
Person.getInfo = function() {
    console.log('Person的静态方法');
}
/* 实例之间共享 */
Person.prototype.eat = function() {
    console.log(this.name + 'eat');
}
/* 继承 */

function Teacher(name, age) {
    Person.call(this, name, age);
}
Teacher.prototype = Object.create(Person.prototype);
Teacher.prototype.constructor = Teacher;

# 接口interface

在面向对象的编程中，接口是一种关于规范的定义，定义了行为和动作的规范。接口规定必须提供某些方法和属性，接口还有不同的类型，包括属性、函数、可索引、类、扩展接口等；

属性接口

/* 属性接口：对json的约束 */ 
function printLabel(
    labelInfo: {
        label: string,
        value: string
    }
):void {
    console.log('printLabel', labelInfo.label);
}
/* 等于，必须包含interface中的属性 */
interface Label {
    label: string,
    value: string,
    key?:number // 可选属性
}
function printLabel2(label: Label):void {
    console.log('printLabel', label.label);
}
interface Config {
    type: string,
    url: string,
    data?: string,
    dataType: string
}
function ajax(config: Config) {
    const xhr = new XMLHttpRequest();
    xhr.open(config.type, config.url, true); // method,url,是否异步
    xhr.send(config.data)；
    xhr.onreadystatechange = function() {
        if(xhr.readyState === 4 && xhr.status === 200) {
            if(config.dataType= === 'json') {
                console.log(JSON.parse(xhr.resposeText));
            } else {
                console.log(xhr.resposeText);
            }
        }
    }
}
ajax({
    type: 'get',
    url: '...',
    data: 'name=1',
    dataType: 'json'
})

函数接口，对入参和返回值进行约束。

interface encrypt {
    (key: string, value: string):string
}
const md5:encrypt = function(key: string, value: string):string {
    return key + value;
}

可索引接口，对数组、对象的约束。不常用。

/* 定义数组 */
const arr1:number[]=[1,2];
const arr2:Array<number> = [3,4];
/* 使用接口 */
interface UserArr{
    [index: number]: string
}
const userArr:UserArr = ['1', '2'];

interface UserObj {
    [index: string]: string;
}
const user:UserObj = {
    name: '1',
    age: '2'
}

类类型接口：对类的约束。和抽象类有点相似，但不是通过继承实现的。在ts中，接口使用implements（中文翻译：贯彻、执行）关键字：

interface Person {

class Teacher implements Person {

# 范型

解决函数、类、接口、方法的复用性，以及对不确定数据类型的支持。

/* 范型，函数入参和返回值的数据类型一致 */
/* T可以用任意大写字母代替 */
function fun<T>(value: T):T {
    return value;
}
/* 调用时，具体数据类型在调用时决定的 */
fun<number>(123);
fun<number>('1'); // 不能通过编译

函数类型接口范型写法：

interface Config {
    (value1: string, value2: string): string
}
const setData:Config = function(value1: string, value2: string): string {
    return value1 + value2;
}
/* 写法一 */
interface Config2 {
    <T>(value: T):T;
}
const setData2:Config2 = function<T>(value: T):T {
    return value
}
setData2<string>('123');// 限定入参必须是string
/* 写法二 */
interface Config2 <T>{
    (value: T):T;
}

function setData2<T>(value: T):T {
    return value
}
const myGetData:Config2<string> = setData2; // 限定入参必须是string
myGetData('123') // 入参必须是string

类的范型：

/* 非范型，一般写法 */
class MinClass {
    public list:number[]=[];
    add(num: number):void {
        this.list.push(num)
    }
    min():number {
        return Math.min(...this.list);
    }
}
/* 类的范型写法 */
/* 范型类，可以同时支持返回不同的数据类型 */
class MinClass2<T> {
    public list:T[]=[];
    add(value: T):void {
        this.list.push(value)
    }
    min():T {
        let minT = this.list[0];
        for(let i = 0; i < this.list.length; i++) {
            /* 字符串和数字使用ascii码进行比较大小 */
            if(minT > this.list[i]) {
                minT = this.list[i];
            }
        }
        return minT;
    }
}
/* 实例化类，并指定T的类型，为number */
const m = new MinClass2<number>();
m.add(1);
/* m.add('2'); 不能通过编译 */
const m2 = new MinClass2<string>();
m2.add('2');

把类作为参数的类型的范型类，使用类作为参数约束数据传入的类型：

/* 数据类 */
class User {
    username: string | undefined;
    password: string | undefined;
}
class MysqlDB {
    add(user: User):boolean {
        return true;
    }
}
/* 实例化数据类 */
const p = new User();
p.username = 'p';
p.password = 'p';

const Db = new MysqlDB();
Db.add(p);
/* ------------------------- */
/* 使用范型，避免MysqlDB重复封装 */
class User {
    username: string | undefined;
    password: string | undefined;
    constructor(username?:string, password?:string) {
        this.username = username;
        this.password = password;
    }
}

class MysqlDB<T> {
    add(info: T):boolean {
        return true;
    }
    update(info: T, id: number):boolean {
        return true;
    }
}
/* 实例化数据类 */
const p = new User();

p.username = 'p';
p.password = 'p';

const p1 = new User('p1', 'p2');
/* 校验数据类型 */
/* new 一个专门用来操作user表的实例，只能传入User的实例数据 */
const userDb = new MysqlDB<User>();

Db.add(p);
Db.add(p1);
Db.add(1); // 编译报错不通过

Db.update(p, 12);

# 模块和命名空间

TS中模块分为内部模块和外部模块；

  • 内部模块，即命名空间；
  • 外部模块，指的是模块有其自身的作用域，在自身作用域里执行；

模块，与ES6的module相同，侧重于代码的复用：

/* 单个导出 */
export const a = 1;

const a = 1;
export { a };

import { a } from '';
/* export defualt */

export default a;

import a from '';

命名空间，在代码量较大的情况下，避免变量、函数命名冲突，将功能相似的函数、类、接口等放到命名空间内。

namespace A {
    export class A {
    }
}
namespace B {
    export class A {
    }
}

const a = new A.A();

# 装饰器

装饰器是一种声明，能够附加给类本身、类的属性、类的方法、类的方法的参数，修改类的行为、扩展类的功能。 包括：类装饰器、类属性装饰器、类方法装饰器、类方法参数装饰器；

也是ES7重要的的标准特性之一，与TS使用方式一样

/* 1.类装饰器，普通装饰器(无法传参) */
function logClass(target: any) {
    conosle.log(target); // params就是被装饰的类
    target.prototype,apiUrl = 'xxxx';
}

@logClass
class HttpClient {
    constructor() {}
    getData() {}
}

const hc = new HttpClient();
hc.apiUrl === 'xxxx'; // true
/* 装饰器工厂(可以传参) */
function logClass(params: string) {
    return function(target: any) {
        console.log(params); // 传入的参数 hello
        console.log(target); // 被装饰的类 HttpClient
    }
}

@logClass('hello')
class HttpClient {
    constructor() {}
    getData() {}
}

/* 重载构造函数，返回一个类的写法 */
function logClass(target: any) {
    return class extends target {
        apiUrl:string = 'xxxx2'; // 此处生效
        getData() {
            console.log(2)
        }
    }
}
/* 属性装饰器，设置默认值 */
function logProperty(params: string) {
    return function(target: any, attr: any) {
        /* target Httpclient ; attr apiUrl; params www.baidu.com */
        target[attr] = params;
    }
}

/* 方法装饰器 */
function logMethod(params: any) {
    return function (target:any, methodName: any, desc: any) {
        console.log(target); // HttpClient
        console.log(methodName); // getData
        console.log(desc); // getData属性描述对象：{value:, writable:, enumerable:, configurable:}
        
        target.apiUrl = 'www.sougou.com';
        target.run = function () {
            console.log('run');
        }

        /* 方法装饰器，修改原方法的参数： */
        /* 1.保存方法，处理参数 */
        const originMethod = desc.value;
        desc.value = function(...args:any[]) {
            args = args.map(vlaue => String(vlaue));
            console.log('装饰器中的方法', args);
            /* 传入处理后的参数，到原方法 */
            originMethod.apply(this, args);
        }
    }
}

/* 方法参数装饰器，装饰方法的参数 */
function logParams(params: any) {
    return function(target: any, methodName: any, paramIndex: any) {
        console.log(params); // xxxx
        console.log(target); // HttpClient
        console.log(methodName); // getData
        console.log(paramIndex); // 0
    }
}

@logClass
class HttpClient {
    @logProperty('www.baidu.com')
    public apiUrl:string | undefined;
    constructor() {
        this.apiUrl = 'xxxx1'
    }
    @logMethod('www.google.com')
    getData(@logParams('xxxx') uuid: any) {
        console.log('HttpClient中的方法');
        console.log(uuid)
    }
}
